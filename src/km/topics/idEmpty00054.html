សម្រាប់អាយុកាលប្រើប្រាស់អាគុយដ៏ល្អបំផុត

  • រក្សាការដំឡើងអាគុយឱ្យមានសុវត្ថិភាព។

  • រក្សាផ្នែកខាងលើនៃអាគុយឱ្យស្អាត និងស្ងួត។

  • រក្សាប៉ូល និងតំណអាគុយឱ្យស្អាត តឹង និងលាបចាហួយប្រេង ឬខ្លាញ់គោលើប៉ូលអាគុយ។

  • លាងជម្រះអេឡិចត្រូលីតដែលកំពប់ចេញពីអាគុយភ្លាមៗជាមួយនឹងសូលុយស្យុងទឹក និងម្សៅសូដាសម្រាប់ដុតនំ។

  • ប្រសិនបើរថយន្តនឹងមិនត្រូវបានប្រើរយៈពេលយូរទេ សូមផ្តាច់ខ្សែអាគុយ។

ការជូនដំណឹង

រថយន្តរបស់អ្នកត្រូវបានបំពាក់ដោយអាគុយថែទាំដោយឥតគិតថ្លៃ។ ប្រសិនបើរថយន្តរបស់អ្នកត្រូវបានបំពាក់ដោយអាគុយដែលមានសញ្ញា LOWER (ទាប) និង UPPER (ខ្ពស់) នៅផ្នែកចំហៀង អ្នកអាចពិនិត្យកម្រិតអេឡិចត្រូលីតបាន។ កម្រិតអេឡិចត្រូលីតគួរតែស្ថិតនៅចន្លោះកម្រិត LOWER (ទាប) និងកម្រិត UPPER (ខ្ពស់)។ ប្រសិនបើកម្រិតអេឡិចត្រូនិចនៅកម្រិតទាប វាចាំបាច់ត្រូវបន្ថែមទឹកដែលបានបិត (គ្មានជាតិរ៉ែ) (មិនត្រូវបន្ថែមអាស៊ីតស៊ុលហ្វួរីក ឬអេឡិចត្រូនិចផ្សេងៗ)។ នៅពេលចាក់បំពេញឡើងវិញ ចូរប្រុងប្រយ័ត្នកុំឱ្យខ្ទាតចំអាគុយ និងសមាសភាគដែលនៅជាប់គ្នានោះ។ ហើយកុំចាក់បំពេញសន្ទះខណ្ឌបន្ទប់អាគុយច្រើនពេក។ វាអាចបណ្តាលកាត់ដោយអាស៊ីតនៅលើផ្នែកផ្សេងទៀត។ ត្រូវប្រាកដថាកម្របទឹកអាស៊ីតអាគុយត្រូវបានរឹតបន្តឹង។
